		--expand-text: '更多';
		--expand-text: '收起';
}
<card-ellipsis min-height="140">
    <div slot="content">
        12121212</br>
        12121212</br>
        12121212</br>
        12121212</br>
        12121212</br>
        12121212</br>

        12121212</br>
